#include<cstdio>
long long B,P,K;
//base ->底数
//p ->指数
long long qpow(int base,int p){
	long long ans=1,tmp=base;//从底数开始乘，不停自乘
	while(p!=0){//指数不是0 
		if(p&1){
			ans=(ans%K*tmp%K)%K;
		} 
		tmp=(tmp%K*tmp%K)%K;//自乘
		p=p>>1;//访问下一位 
	} 
	ans=ans%K;
	return ans;
} 
int main(){
	scanf("%lld%lld%lld",&B,&P,&K);
	long long ans=qpow(B,P);
	ans=ans%K;
	printf("%lld^%lld mod %lld=%lld",B,P,K,ans);
	return 0;
}
